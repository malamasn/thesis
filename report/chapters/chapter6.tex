\begin{itemize}
    \setlength\itemsep{-0.2em}
    \item επιτυχία εντοπισμού κάθε πόρτας του χώρου
    \item επιλογή βέλτιστης σειράς επίσκεψης των δωματίων
    \item πλήρης κάλυψη του χάρτη
\end{itemize}
